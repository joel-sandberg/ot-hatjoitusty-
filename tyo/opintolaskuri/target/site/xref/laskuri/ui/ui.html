72      public void welcome(Stage window, account acc) {
73          Label welcomeText = new Label("Welcome, " + acc.getName() + "!");
74          
75          Label credits = new Label("Your current credits: " + acc.getCredit());
76          Label untilBc = new Label("Credits until bachelor: " + (180 - acc.getCredit()));
77          Label untilMs = new Label("Credits until masters: " + (360 - acc.getCredit()));
78          Label nConame = new Label("Name of completed course:");
79          TextField newCourse = new TextField();
80          Label nCoamount = new Label("Amount of credits for completed course:");
81          TextField newCredit = new TextField();
82          Button addnewCredits = new Button("Add completed course");
83          Label comp = new Label("Courses you have completed:");
84          Label compC = new Label("");
85          VBox welcomeSet = new VBox();
86          welcomeSet.setPrefSize(300, 180);
87          
88          welcomeSet.getChildren().addAll(welcomeText, credits, untilBc, untilMs, nConame, newCourse, nCoamount, newCredit, addnewCredits, comp, compC);
89          welcomeSet.setAlignment(Pos.TOP_CENTER);
90  
91           addnewCredits.setOnAction((event) -> {
92               
93              int add = Integer.parseInt(newCredit.getText());
94              acc.addCredit(add);
95              String text = newCourse.getText();
96              newCredit.setText("");
97              newCourse.setText("");
98              credits.setText("Your current credits: " + acc.getCredit());
99              untilBc.setText("Credits until bachelor: " + (180 - acc.getCredit()));
100             untilMs.setText("Credits until masters: " + (360 - acc.getCredit()));
101             
102             compC.setText(compC.getText() + " " + text + ", " + add + " credits;");
103             compC.setPrefWidth(250);
104             compC.setWrapText(true);
105         });
106         
107         Scene loggedInWindow = new Scene(welcomeSet);
108         window.setScene(loggedInWindow);
109     }
110     public void login(Stage window) {
111         window.setTitle("Studies");
112 //luodaan alkuikkunan asetukset
113         BorderPane set = new BorderPane();
114 
115         
116         texts.setSpacing(20);
117         TextField logName = new TextField();
118         PasswordField pWord = new PasswordField();
119         Label header = new Label("Study Calculator pro 3000");
120         Label askPWord = new Label("Password");
121         Label askLogin = new Label("Log in");
122         Label erText = new Label("");
123         texts.getChildren().addAll(header, askLogin, logName, askPWord, pWord, erText);
124 
125         
126         buttons.setSpacing(20);
127         Button logIn = new Button("Log in");
128         Button createAcc = new Button("Create a new account");
129         buttons.getChildren().add(logIn);
130         buttons.getChildren().add(createAcc);
131 
132         
133 //laitetaan nappi skulaa
134         logIn.setOnAction((event) -> {
135             this.acc = new account(logName.getText(), pWord.getText());
136             if (!logName.getText().trim().equals("admin")) {
137                 erText.setText("This account does not exist.");
138                 return;
139             }
140             if (!pWord.getText().trim().equals("123")) {
141                 erText.setText("Unknown password.");
142                 return;
143             }
144             welcome(window, this.acc);
145         });
146 //laitetaa toka nappi skulaa
147         createAcc.setOnAction((event) -> {
148             
149             crtAcc(window);
150         });
151 //roska käyntiin
152         set.setCenter(texts);
153         set.setBottom(buttons);
154         Scene start = new Scene(set);
155         window.setScene(start);
156 
157         window.show();
158     }
